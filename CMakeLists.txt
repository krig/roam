cmake_minimum_required(VERSION 2.8.12)
project(roam)
set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -O0 -g -Wall -std=c11")
include(FindPkgConfig)
pkg_search_module(GL REQUIRED gl)
pkg_search_module(GLEW REQUIRED glew)
pkg_search_module(SDL2 REQUIRED sdl2)
include_directories("src")
include_directories("stb")
include_directories(${GL_INCLUDE_DIRS})
include_directories(${GLEW_INCLUDE_DIRS})
include_directories(${SDL2_INCLUDE_DIRS})
file(GLOB_RECURSE sources "src/*.c")
add_executable(roam ${sources})
target_link_libraries(roam ${GL_LIBRARIES})
target_link_libraries(roam ${GLEW_LIBRARIES})
target_link_libraries(roam ${SDL2_LIBRARIES})
target_link_libraries(roam m)
